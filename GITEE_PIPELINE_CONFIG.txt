version: '1.0'
name: build-and-package
displayName: 构建和打包
triggers:
  push:
    branches:
      - master
      - main

variables:
  JAVA_VERSION: '17'

jobs:
  build-windows:
    name: windows-build
    displayName: Windows平台打包
    runsOn: windows-latest
    steps:
      - uses: checkout@v1
      
      - name: Setup Java
        uses: setup-java@v1
        with:
          java: ${{ env.JAVA_VERSION }}
      
      - name: Build JAR
        run: |
          mvn clean package -DskipTests
        shell: powershell
      
      - name: Package with jpackage
        run: |
          $jarFile = Get-ChildItem -Path target -Filter "batchfile_app-*.jar" -Exclude "*.original" | Select-Object -First 1
          jpackage --input target `
                   --name "批量文件夹重命名工具" `
                   --main-jar $jarFile.Name `
                   --main-class com.rt.batchfile_app.BatchfileAppApplication `
                   --dest target/dist `
                   --type app-image `
                   --app-version 1.0.0 `
                   --vendor RT `
                   --java-options "--add-modules" `
                   --java-options "javafx.controls,javafx.fxml"
        shell: powershell
      
      - uses: upload@v1
        with:
          name: windows-app
          path:
            - target/dist/批量文件夹重命名工具/
